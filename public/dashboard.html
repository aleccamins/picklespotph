const API = "http://localhost:5000"; // change if deployed // LOAD COURTS async function loadCourts() { const res = await fetch(`${API}/courts`); const data = await res.json(); const container = document.getElementById("courts"); container.innerHTML = ""; data.forEach(court => { container.innerHTML += `

${court.name}

${court.location}

`; }); } // LOAD BOOKINGS async function loadBookings() { const res = await fetch(`${API}/bookings`); const data = await res.json(); const container = document.getElementById("bookings"); container.innerHTML = ""; data.forEach(b => { container.innerHTML += `

Court: ${b.courtName}

User: ${b.user}

Date: ${b.date}

Status: ${b.status}

`; }); } // APPROVE async function approveBooking(id) { await fetch(`${API}/approve/${id}`, { method: "POST" }); loadBookings(); } // REJECT async function rejectBooking(id) { await fetch(`${API}/reject/${id}`, { method: "POST" }); loadBookings(); } // INIT loadCourts(); loadBookings();
